\subtitle{File Commands}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    ls  & Directory listing\\
    ls -al& Formatted list all\\
    cd [dir]& Change directory to \tr{dir} \\
    cd -    & Change to previous directory\\
    pwd & Show current directory\\
    mkdir dir & Create a directory \tr{dir}\\
    rm [-r][-f] file & Delete \tr{file} \tr{f}orce/\tr{r}ecursively\\
    cp file1 file2 & Copy \tr{file1} to \tr{file2} \\
    mv file1 file2 & Rename or move \tr{file1} to \tr{file2}\\
    ln [-s] file link & Create [symbolic] link\\
    touch file & Create or update \tr{file}\\
    cat $>$ file & Put standard input into \tr{file}\\
    more file & Show \tr{file} page by page\\
    head -n 10 file & Show first 10 lines of \tr{file}\\
    tail -n 5 file & Show last 5 lines of \tr{file}\\
    tail -f file & \tr{F}ollow the contents of \tr{file}
\end{tabular}
}

\subtitle{Searching}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    grep pattern files & Search \tr{pattern} in \tr{files}\\
    grep -r pattern dir & Search \tr{r}ecursively in \tr{dir}\\
    find dir -name '*.c' & Find files with suffix .c\\
    locate file & Find instances of \tr{ file}
\end{tabular}
}

\subtitle{Installation}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    ./configure & configure from source\\
    make       &  build from source\\
    make install & install \\
    rpm    & RedHat package manager\\
    dpkg   & Debian package manager\\
           & -i \tr{pkg.deb} \tr{i}nstall/upgrade\\
           & -r \tr{pkg}  \tr{r}emove\\
    apt-get & repository package manager
\end{tabular}
}

\subtitle{Compression}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    tar & Create/extract a tar file\\
        &{\bf f -- file}\\
        &{\bf c -- create/compression}\\
        &{\bf ~~z -- gzip zipped}\\
        &{\bf ~~j -- bz2 zipped}\\
        &{\bf ~~J -- xz zipped}\\
        &{\bf x -- extract}\\
        & Examples:\\
        & tar xf foo.tar.bz2\\
        & tar zcf foo.tar.gz dir\\
    gzip file & Compress \tr{file} to \tr{file.gz}\\
    gzip -d file.gz & Decompress \tr{file.gz}
\end{tabular}
}

\subtitle{System Info}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    date  & Show the current \tr{date} and time\\
    cal   & Show this month's \tr{cal}endar\\
    uptime & Show current \tr{uptime}\\
    w  & Display who is online\\
    whoami & Who you are logged in as\\
    uname -a & Show kernel information\\
    man command & Show the \tr{man}ual for \tr{command}\\
    df & Show disk usage\\
    du & Show directory space usage\\
    free & Show memory and swap usage\\
    whereis app & Show possible locations of \tr{app}
\end{tabular}
}

\subtitle{File Permissions}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    chmod & Change the permissions(rwx) of file\\
	  & {\bf 4 -- read(r)}\\
	  & {\bf 2 -- write(w)}\\
	  & {\bf 1 -- execute(x)}\\
	  & Examples:\\
	  & chmod 777 -- RWX for all\\
	  & chmod 755 -- owner RWX , others R-W\\
    chown & Change owner@group of file(s)
\end{tabular}
}

\subtitle{Process Management}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    ps & Display active processes\\
    top & Display running processes\\
    kill pid & Kill process by \tr{pid}(number)\\
    killall proc & Kill all processes named \tr{proc}\\
    bg & Send a job into background \\
    fg job & Bring \tr{job} to the foreground
\end{tabular}
}

\subtitle{Network}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    ping host & check the \tr{host} connection\\
    whois domain & get whois for \tr{domain}\\
    dig domain & get DNS information\\
    dig -x host & reverse lookup \tr{host}\\
    wget URL:file & download \tr{file} from network\\
    wget -c file & continue a stopped download\\
\end{tabular}
}

\subtitle{Shortcuts}
\shadowbox{
\begin{tabular}{>{\columncolor{gray2}}ll}
    $^\wedge$C & Halt the current command\\
    $^\wedge$Z & Stop the current command, \\
          & cab be resumed with \tr{fg} or \tr{bg} \\
    $^\wedge$D &  Exit current shell/session\\
    $^\wedge$W &  Erase one word in the current line\\
    $^\wedge$U & Erase the whole line\\
    !!    &   Repeat the last command\\
    $*$   & \tr{USE WITH EXTREME CAUTION}
\end{tabular}
}
